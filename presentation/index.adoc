=== What is docker-compose?

[%step]
* multi-container docker applications
* defined through YAML files

[state=introduction]
=== !

....
version: '2'
services:
  web:
    build: .
    ports:
    - "5000:5000"
    volumes:
    - .:/code
    - logvolume01:/var/log
    links:
    - postgres
  postgres:
    image: postgres:latest
    expose:
    - "5432"
volumes:
  logvolume01: {}
....

== SSL Termination with Nginx proxy

....
frontend-service:
    image: frontend-service:latest
    environment:
      VIRTUAL_HOST: frontend-service
      CERT_NAME: server
      VIRTUAL_PORT: 9443
      VIRTUAL_PROTO: https
      KEYSTORE_PASSWORD: suchsecret
      KEYSTORE_PATH: /opt/docker/conf/manysecrets.keystore
      HTTPS_PORT: 8443
    volumes:
      - ./cacerts:/etc/ssl/certs/java/cacerts:ro
    expose:
      - "8443"
    depends_on:
      - parcel-service
      - proxy
    links:
      - proxy:parcel-service
....
